WITH RankedDates AS (
    SELECT 
        A,
        B,
        [Date],
        ROW_NUMBER() OVER (ORDER BY [Date]) - 
        ROW_NUMBER() OVER (PARTITION BY A, B ORDER BY [Date]) AS GroupID
    FROM 
        YourTable
),
GroupedDates AS (
    SELECT 
        A,
        B,
        MIN([Date]) AS StartDate,
        MAX([Date]) AS EndDate
    FROM 
        RankedDates
    GROUP BY 
        A, B, GroupID
)
SELECT 
    A,
    B,
    StartDate,
    EndDate
FROM 
    GroupedDates
WHERE 
    DATEDIFF(day, StartDate, EndDate) >= 1
ORDER BY 
    StartDate, A, B;
