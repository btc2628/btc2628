	DECLARE @date as DATE = '2024-04-30'

declare @weekEndDate as date = CASE WHEN DATEPART(WEEKDAY, @date) < 5 THEN DATEADD(DAY, -1 - DATEPART(WEEKDAY, @date), CAST(@date as DATE))
									WHEN DATEPART(WEEKDAY, @date) > 5 THEN DATEADD(DAY, -1 - DATEPART(WEEKDAY, @date), CAST(@date as DATE))
									ELSE DATEADD(DAY, 7, CAST(@date as DATE))
